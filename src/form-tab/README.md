### FormTab.createFormTab

```typescript
type ActiveKey = string | number;

interface createFormTab {
  (defaultActiveKey?: ActiveKey): IFormTab;
}

interface IFormTab {
  //激活主键
  activeKey: ActiveKey;
  //设置激活主键
  setActiveKey(key: ActiveKey): void;
}
```
