### FormStep.createFormStep

```typescript
interface createFormStep {
  (current?: number): IFormStep;
}

interface IFormTab {
  //当前索引
  current: number;
  //是否允许向后
  allowNext: boolean;
  //是否允许向前
  allowBack: boolean;
  //设置当前索引
  setCurrent(key: number): void;
  //提交表单
  submit: Form["submit"];
  //向后
  next(): void;
  //向前
  back(): void;
}
```
